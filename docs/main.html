    global.PACKAGE = PACKAGE

Github = require "github"

api = Github().api

module.exports = GistEval =
  exec: (id) ->
    api("gists/#{id}")
    .then (data) ->
      files = data.files

      # Eval all .js
      Object.keys(files).forEach (name) ->
        if name.match(/\.js$/)
          Function(files[name].content)()
